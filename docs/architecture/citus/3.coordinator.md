  NixOS Citus Module (Complete)

  # infra/nixos/modules/services/citus.nix
  { config, lib, pkgs, ... }:

  with lib;

  let
    cfg = config.services.citus;
  in {
    options.services.citus = {
      enable = mkEnableOption "Citus distributed PostgreSQL";

      role = mkOption {
        type = types.enum [ "coordinator" "worker" "coordinator-worker" ];
        description = "Node role in Citus cluster";
      };

      coordinatorHost = mkOption {
        type = types.str;
        default = "";
        description = "Coordinator hostname (for workers)";
      };

      workers = mkOption {
        type = types.listOf types.str;
        default = [];
        description = "Worker hostnames (for coordinator)";
      };
    };

    config = mkIf cfg.enable {
      services.postgresql = {
        enable = true;
        package = pkgs.postgresql_16;
        extensions = ps: [ ps.citus ];

        settings = {
          shared_preload_libraries = "citus";
          listen_addresses = "*";
          max_connections = 300;

          # Performance
          shared_buffers = "2GB";
          effective_cache_size = "6GB";
          work_mem = "32MB";
          maintenance_work_mem = "512MB";
        };

        authentication = ''
          local all all peer
          host all all 100.64.0.0/10 scram-sha-256
          host all all 10.0.0.0/8 scram-sha-256
        '';

        initialScript = pkgs.writeText "citus-init.sql" ''
          CREATE EXTENSION IF NOT EXISTS citus;
        '';
      };

      networking.firewall.interfaces."tailscale0" = {
        allowedTCPPorts = [ 5432 ];
      };
    };
  }
